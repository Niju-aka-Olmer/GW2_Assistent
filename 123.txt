// src/entities/item/ui/ItemCard.tsx
import { useState } from 'react';
import { ItemTooltip } from '@/widgets/ItemTooltip';
import type { Item } from '../model/types';

interface ItemCardProps {
  item: Item;
  count?: number;
}

export function ItemCard({ item, count }: ItemCardProps) {
  const [showTooltip, setShowTooltip] = useState(false);
  
  // Определяем цвет рамки по редкости
  const rarityColors: Record<string, string> = {
    Junk: 'border-gray-400',
    Basic: 'border-white',
    Fine: 'border-blue-500',
    Masterwork: 'border-green-600',
    Rare: 'border-yellow-500',
    Exotic: 'border-orange-500',
    Ascended: 'border-purple-600',
    Legendary: 'border-red-600'
  };
  
  const iconUrl = item.icon || `https://render.guildwars2.com/file/${item.icon_file_signature}/${item.icon_file_id}.png` [citation:9];
  
  return (
    <div 
      className={`
        relative w-16 h-16 rounded-lg border-2 
        ${rarityColors[item.rarity] || 'border-gray-600'}
        bg-gray-800 hover:shadow-lg transition-shadow
        cursor-pointer overflow-hidden
      `}
      onMouseEnter={() => setShowTooltip(true)}
      onMouseLeave={() => setShowTooltip(false)}
    >
      <img 
        src={iconUrl} 
        alt={item.name}
        className="w-full h-full object-contain p-1"
        loading="lazy"
      />
      
      {count && count > 1 && (
        <div className="absolute bottom-0 right-0 bg-black/70 text-white text-xs px-1 rounded-tl">
          {count}
        </div>
      )}
      
      <ItemTooltip item={item} visible={showTooltip} />
    </div>
  );
}

# backend/main.py
from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse
import httpx
from typing import List, Optional
import asyncio
from cachetools import TTLCache

app = FastAPI(title="GW2 Assistant API")

# CORS для фронтенда
app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000", "http://127.0.0.1:3000"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Кэш для предметов (TTL = 1 час)
items_cache = TTLCache(maxsize=10000, ttl=3600)

# API ключи пользователей (в реальном проекте - в БД)
user_api_keys = {}  # session_id -> gw2_api_key

# Базовые URL
GW2_API_URL = "https://api.guildwars2.com/v2"
RENDER_URL = "https://render.guildwars2.com/file"

@app.post("/api/auth")
async def set_api_key(api_key: str, session_id: str):
    """Сохранение API ключа пользователя"""
    # Проверяем ключ
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/tokeninfo",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        
        if resp.status_code != 200:
            raise HTTPException(status_code=401, detail="Invalid API key")
        
        permissions = resp.json().get("permissions", [])
        
        # Проверяем необходимые права [citation:3]
        required = ["account", "characters", "inventories", "builds"]
        missing = [p for p in required if p not in permissions]
        
        if missing:
            raise HTTPException(
                status_code=403, 
                detail=f"Missing permissions: {missing}"
            )
    
    user_api_keys[session_id] = api_key
    return {"status": "ok", "permissions": permissions}

@app.get("/api/characters")
async def get_characters(session_id: str):
    """Получить список персонажей"""
    api_key = user_api_keys.get(session_id)
    if not api_key:
        raise HTTPException(status_code=401, detail="Not authenticated")
    
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/characters",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        
        if resp.status_code != 200:
            raise HTTPException(status_code=resp.status_code, detail="GW2 API error")
        
        characters = resp.json()
        
        # Получаем дополнительную инфу о каждом персонаже
        result = []
        for name in characters:
            char_resp = await client.get(
                f"{GW2_API_URL}/characters/{name}",
                headers={"Authorization": f"Bearer {api_key}"}
            )
            char_data = char_resp.json()
            result.append({
                "name": name,
                "level": char_data.get("level"),
                "profession": char_data.get("profession"),
                "race": char_data.get("race"),
                "age": char_data.get("age"),
                "created": char_data.get("created")
            })
        
        return result

@app.get("/api/characters/{name}/build")
async def get_character_build(name: str, session_id: str):
    """Получить билд персонажа (специализации и скиллы)"""
    api_key = user_api_keys.get(session_id)
    if not api_key:
        raise HTTPException(status_code=401, detail="Not authenticated")
    
    async with httpx.AsyncClient() as client:
        # Специализации
        spec_resp = await client.get(
            f"{GW2_API_URL}/characters/{name}/specializations",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        specializations = spec_resp.json()
        
        # Навыки
        skills_resp = await client.get(
            f"{GW2_API_URL}/characters/{name}/skills",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        skills = skills_resp.json()
        
        # Экипировка
        equip_resp = await client.get(
            f"{GW2_API_URL}/characters/{name}/equipment",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        equipment = equip_resp.json()
        
        # Обогащаем предметы данными с русским языком [citation:2][citation:8]
        enriched_equipment = []
        for item in equipment.get("equipment", []):
            item_data = await get_item_details(item["id"], lang="ru")
            enriched_equipment.append({
                **item,
                "details": item_data
            })
        
        return {
            "name": name,
            "specializations": specializations,
            "skills": skills,
            "equipment": enriched_equipment
        }

@app.get("/api/characters/{name}/inventory")
async def get_character_inventory(name: str, session_id: str):
    """Получить инвентарь персонажа с деталями на русском"""
    api_key = user_api_keys.get(session_id)
    if not api_key:
        raise HTTPException(status_code=401, detail="Not authenticated")
    
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/characters/{name}/inventory",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        inventory = resp.json()
        
        # Собираем все ID предметов для массового запроса
        item_ids = set()
        bags = inventory.get("bags", [])
        
        for bag in bags:
            if bag and "inventory" in bag:
                for slot in bag["inventory"]:
                    if slot and "id" in slot:
                        item_ids.add(slot["id"])
        
        # Получаем детали предметов (с русской локализацией) [citation:3][citation:8]
        items_details = await get_items_details(list(item_ids), lang="ru")
        
        # Получаем цены [citation:3]
        prices = await get_items_prices(list(item_ids))
        
        # Собираем результат
        result = {
            "bags": [],
            "prices": prices
        }
        
        for bag in bags:
            if bag and "inventory" in bag:
                enriched_slots = []
                for slot in bag["inventory"]:
                    if slot and "id" in slot:
                        slot_data = {
                            "id": slot["id"],
                            "count": slot.get("count", 1),
                            "slot": slot.get("slot"),
                            "binding": slot.get("binding"),
                            "details": items_details.get(slot["id"]),
                            "price": prices.get(slot["id"])
                        }
                        enriched_slots.append(slot_data)
                
                result["bags"].append({
                    "id": bag.get("id"),
                    "size": bag.get("size"),
                    "inventory": enriched_slots
                })
        
        return result

async def get_item_details(item_id: int, lang: str = "ru") -> dict:
    """Получить детали предмета с кэшем [citation:2][citation:5]"""
    cache_key = f"{item_id}_{lang}"
    
    if cache_key in items_cache:
        return items_cache[cache_key]
    
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/items/{item_id}",
            params={"lang": lang}  # локализация на русский [citation:8]
        )
        
        if resp.status_code == 200:
            data = resp.json()
            
            # Формируем URL иконки [citation:9]
            if "icon" in data:
                # иконка уже в полном URL в v2
                pass
            elif "icon_file_id" in data and "icon_file_signature" in data:
                data["icon"] = f"{RENDER_URL}/{data['icon_file_signature']}/{data['icon_file_id']}.png"
            
            items_cache[cache_key] = data
            return data
        
        return {"id": item_id, "name": f"Unknown [{item_id}]"}

async def get_items_details(item_ids: List[int], lang: str = "ru") -> dict:
    """Массовое получение деталей предметов (до 200 за раз) [citation:7]"""
    result = {}
    
    for i in range(0, len(item_ids), 200):
        batch = item_ids[i:i+200]
        ids_param = ",".join(str(id) for id in batch)
        
        async with httpx.AsyncClient() as client:
            resp = await client.get(
                f"{GW2_API_URL}/items",
                params={"ids": ids_param, "lang": lang}
            )
            
            if resp.status_code == 200:
                items = resp.json()
                for item in items:
                    # Формируем URL иконки если нужно
                    if "icon" not in item and "icon_file_id" in item:
                        item["icon"] = f"{RENDER_URL}/{item['icon_file_signature']}/{item['icon_file_id']}.png"
                    result[item["id"]] = item
    
    return result

async def get_items_prices(item_ids: List[int]) -> dict:
    """Получить цены на торговом посту [citation:3]"""
    result = {}
    
    for i in range(0, len(item_ids), 200):
        batch = item_ids[i:i+200]
        ids_param = ",".join(str(id) for id in batch)
        
        async with httpx.AsyncClient() as client:
            resp = await client.get(
                f"{GW2_API_URL}/commerce/prices",
                params={"ids": ids_param}
            )
            
            if resp.status_code == 200:
                prices = resp.json()
                for price in prices:
                    result[price["id"]] = {
                        "buy_price": price.get("buys", {}).get("unit_price"),
                        "sell_price": price.get("sells", {}).get("unit_price"),
                        "buy_quantity": price.get("buys", {}).get("quantity"),
                        "sell_quantity": price.get("sells", {}).get("quantity")
                    }
        # Небольшая задержка между батчами
        await asyncio.sleep(0.1)
    
    return result

// Функция конвертации coin value в золото/серебро/медь
function formatCoin(value: number): string {
  const gold = Math.floor(value / 10000);
  const silver = Math.floor((value % 10000) / 100);
  const copper = value % 100;
  
  const parts = [];
  if (gold > 0) parts.push(`${gold} 💛`);
  if (silver > 0) parts.push(`${silver} 🥈`);
  if (copper > 0 || parts.length === 0) parts.push(`${copper} 🥉`);
  
  return parts.join(' ');
}

from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
import httpx
from cachetools import TTLCache
from typing import List, Optional, Dict
import asyncio

app = FastAPI(title="GW2 Assistant API")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000", "http://127.0.0.1:3000"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Кэши
items_cache = TTLCache(maxsize=10000, ttl=3600)
prices_cache = TTLCache(maxsize=5000, ttl=300)

# Хранилище ключей (в продакшене использовать БД)
user_keys: Dict[str, str] = {}

GW2_API_URL = "https://api.guildwars2.com/v2"
RENDER_URL = "https://render.guildwars2.com/file"

@app.post("/api/auth")
async def set_api_key(api_key: str, session_id: str):
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/tokeninfo",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        if resp.status_code != 200:
            raise HTTPException(401, "Invalid API key")
        
        perms = resp.json().get("permissions", [])
        required = ["account", "characters", "inventories", "builds"]
        missing = [p for p in required if p not in perms]
        
        if missing:
            raise HTTPException(403, f"Missing: {missing}")
    
    user_keys[session_id] = api_key
    return {"status": "ok", "permissions": perms}

@app.get("/api/characters")
async def get_characters(session_id: str):
    api_key = user_keys.get(session_id)
    if not api_key:
        raise HTTPException(401, "Not authenticated")
    
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"{GW2_API_URL}/characters",
            headers={"Authorization": f"Bearer {api_key}"}
        )
        characters = resp.json()
        
        result = []
        for name in characters:
            char_resp = await client.get(
                f"{GW2_API_URL}/characters/{name}",
                headers={"Authorization": f"Bearer {api_key}"}
            )
            data = char_resp.json()
            result.append({
                "name": name,
                "level": data.get("level"),
                "profession": data.get("profession"),
                "race": data.get("race")
            })
        return result

@app.get("/api/characters/{name}/inventory")
async def get_inventory(name: str, session_id: str):
    api_key = user_keys.get(session_id)

// Компонент тултипа
function Tooltip({ item, visible, position }: any) {
    if (!visible) return null
    
    const rarityColors: any = {
        Junk: '#888', Basic: '#fff', Fine: '#3399ff',
        Masterwork: '#33cc33', Rare: '#ffcc00', Exotic: '#ff9900',
        Ascended: '#cc33ff', Legendary: '#ff3300'
    }
    
    return (
        <div className="tooltip" style={{ top: position.y, left: position.x }}>
            <h4 style={{ color: rarityColors[item.rarity] || '#fff' }}>{item.name}</h4>
            {item.type && <p>Тип: {item.type}</p>}
            {item.level > 0 && <p>Уровень: {item.level}</p>}
            {item.description && <p><em>{item.description}</em></p>}
            <p style={{ marginTop: 8, color: '#888' }}>ID: {item.id}</p>
        </div>
    )
}

// Компонент карточки предмета
function ItemCard({ item, count }: any) {
    const [showTooltip, setShowTooltip] = useState(false)
    const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 })
    
    const handleMouseEnter = (e: React.MouseEvent) => {
        setTooltipPos({ x: e.clientX + 15, y: e.clientY + 15 })
        setShowTooltip(true)
    }
    
    const rarityClass = `rarity-${item.rarity || 'Basic'}`
    
    return (
        <div 
            className={`item-card ${rarityClass}`}
            onMouseEnter={handleMouseEnter}
            onMouseLeave={() => setShowTooltip(false)}
        >
            {item.icon ? (
                <img src={item.icon} alt={item.name} className="item-icon" />
            ) : (
                <div className="item-icon" style={{ background: '#2a2a4a', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                    📦
                </div>
            )}
            {count && count > 1 && <span className="item-count">{count}</span>}
            <Tooltip item={item} visible={showTooltip} position={tooltipPos} />
        </div>
    )
}

    const [inventory, setInventory] = useState<any>(null)
    const [bank, setBank] = useState<any>(null)
    const [loading, setLoading] = useState(false)

    const [recommendations, setRecommendations] = useState('')
    const [deepseekKey, setDeepseekKey] = useState('')

    // Авторизация
    const handleAuth = async () => {
        setLoading(true)
        try {
            await api.post('/auth', { api_key: apiKey, session_id: sessionId })
            setIsAuth(true)
            await loadCharacters()
        } catch (err: any) {
            alert('Ошибка: ' + (err.response?.data?.detail || 'Неверный ключ'))
        }
        setLoading(false)
    }

    // Загрузка персонажей
    const loadCharacters = async () => {
        const res = await api.get('/characters', { params: { session_id: sessionId } })
        setCharacters(res.data)
    }

    // Загрузка инвентаря
    const loadInventory = async (charName: string) => {
        setLoading(true)
        const res = await api.get(`/characters/${charName}/inventory`, { params: { session_id: sessionId } })
        
        // Обогащаем предметы названиями и иконками
        const enrichedItems: any[] = []
        for (const bag of res.data.bags || []) {
            if (bag?.inventory) {
                for (const slot of bag.inventory) {
                    if (slot?.id) {
                        try {
                            const itemRes = await api.get(`/items/${slot.id}`)
                            enrichedItems.push({
                                ...slot,
                                details: itemRes.data,
                                name: itemRes.data.name,
                                icon: itemRes.data.icon,
                                rarity: itemRes.data.rarity,
                                type: itemRes.data.type,
                                level: itemRes.data.level,
                                description: itemRes.data.description
                            })
                        } catch { }
                    }
                }
            }
        }
        setInventory(enrichedItems)
        setLoading(false)
    }

    // Загрузка банка
    const loadBank = async () => {
        setLoading(true)
        const res = await api.get('/account/bank', { params: { session_id: sessionId } })
        
        const enrichedItems: any[] = []
        for (const slot of res.data) {
            if (slot?.id) {
                try {
                    const itemRes = await api.get(`/items/${slot.id}`)
                    enrichedItems.push({
                        ...slot,
                        details: itemRes.data,
                        name: itemRes.data.name,
                        icon: itemRes.data.icon,
                        rarity: itemRes.data.rarity,
                        type: itemRes.data.type
                    })
                } catch { }
            }
        }

    }

    const handleAnalyze = async () => {

            return

            const res = await api.post('/deepseek/analyze', 
                { name: selectedChar?.name, profession: selectedChar?.profession, deepseek_key: deepseekKey },
                { params: { deepseek_key: deepseekKey } }
            )
            setRecommendations(res.data)

        }
        setLoading(false)
    }
